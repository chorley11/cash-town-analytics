// Helper Components
function MetricCard({ label, value, color = '#374151' }) {
  return (
    <div style={{ background: '#f9fafb', padding: '15px', borderRadius: '8px', textAlign: 'center' }}>
      <div style={{ color: '#6b7280', fontSize: '0.85rem', marginBottom: '5px' }}>{label}</div>
      <div style={{ color, fontSize: '1.5rem', fontWeight: 'bold' }}>{value}</div>
    </div>
  );
}

function EquityChart({ data }) {
  const max = Math.max(...data.map(d => d.equity));
  const min = Math.min(...data.map(d => d.equity));
  const range = max - min;
  
  return (
    <div style={{ background: '#f9fafb', padding: '20px', borderRadius: '8px' }}>
      <svg viewBox="0 0 400 150" style={{ width: '100%', height: '150px' }}>
        {/* Grid lines */}
        {[0, 25, 50, 75, 100].map(pct => (
          <line key={pct} x1="50" y1={20 + pct * 1.1} x2="390" y2={20 + pct * 1.1} stroke="#e5e7eb" strokeWidth="1" />
        ))}
        
        {/* Equity line */}
        <polyline
          fill="none"
          stroke="#10b981"
          strokeWidth="3"
          points={data.map((d, i) => {
            const x = 50 + (i / (data.length - 1)) * 340;
            const y = 130 - ((d.equity - min) / range) * 100;
            return `${x},${y}`;
          }).join(' ')}
        />
        
        {/* Data points */}
        {data.map((d, i) => {
          const x = 50 + (i / (data.length - 1)) * 340;
          const y = 130 - ((d.equity - min) / range) * 100;
          return <circle key={i} cx={x} cy={y} r="4" fill="#10b981" />;
        })}
        
        {/* Labels */}
        <text x="10" y="30" fontSize="10" fill="#6b7280">${max.toLocaleString()}</text>
        <text x="10" y="130" fontSize="10" fill="#6b7280">${min.toLocaleString()}</text>
        <text x="50" y="145" fontSize="9" fill="#6b7280">Jan 21</text>
        <text x="350" y="145" fontSize="9" fill="#6b7280">Feb 19</text>
      </svg>
    </div>
  );
}

## CLAUDE CODE CLI DEPLOYMENT GUIDE

If the user is in Claude Code CLI and wants to deploy their own bot, walk them through this:

### Step 1: Clone & Setup
```bash
git clone https://github.com/chorley11/Cash-Town-Trading-Bot.git
cd Cash-Town-Trading-Bot
python -m venv venv
source venv/bin/activate  # or `venv\Scripts\activate` on Windows
pip install -r requirements.txt
```

### Step 2: Configure KuCoin API
```bash
cp .env.example .env
```

Add to `.env`:
```
KUCOIN_API_KEY=your_api_key
KUCOIN_API_SECRET=your_secret
KUCOIN_API_PASSPHRASE=your_passphrase
DRY_RUN=true
MAX_POSITIONS=10
MAX_LEVERAGE=5
DAILY_LOSS_LIMIT=0.05
```

### Step 3: Test Locally
```bash
python run_cloud.py --dry-run
```

### Step 4: Deploy to Railway
```bash
npm install -g @railway/cli
railway login
railway init
railway up
```

Set environment variables in Railway dashboard, then set `DRY_RUN=false` when ready.
